% ── ABSTRACT ──────────────────────────────────────────────────
\begin{abstract}
We study the integer sequence $T_k = 6 \cdot 2^k + 1$ (OEIS A004119)
and establish a complete theory of its modular periodicity. For each
positive integer~$m$, we define the OMARWA period $P(m)$ as the minimal
period of the sequence $T_k \bmod m$ and prove the \textbf{Core Reduction
Theorem}: $P(m) = \ord_{m'}(2)$ where $m' = m / \gcd(m, 6)$. We derive
\textbf{fractal period laws} $P(p^n) = d_0 \cdot p^{n-1}$ for primes
$p \nmid 6$ under the hypothesis that $p$ is not a Wieferich prime, where
$d_0 = \ord_p(2)$, and prove a \textbf{Product Theorem} via the Chinese
Remainder Theorem: $P(m_1 m_2) = \lcm(P(m_1), P(m_2))$ for coprime
$m_1, m_2$. These results yield a \textbf{super-period}
$L = \lcm(P(9), P(11)) = 30$, derived independently through three
distinct paths. We further establish a \textbf{GCD Law}, a
\textbf{palindromic involution} on residue sequences, and a
\textbf{Pascal--Sierpi\'nski--Fibonacci bridge} connecting modular
periodicity to combinatorial and fractal structures. All principal
theorems are formally verified in Lean~4 with Mathlib
(956~theorems~+~lemmas, zero \texttt{sorry} axioms, 28~modules;
source: \href{https://github.com/omarwa622/OmarwaCore}{github.com/omarwa622/OmarwaCore}).
\end{abstract}

\section{Introduction}
% ══════════════════════════════════════════════════════════════

The sequence $T_k = 6 \cdot 2^k + 1$ for $k \geq 0$ produces the terms
$7, 13, 25, 49, 97, 193, 385, \ldots$ (OEIS A004119, shifted:
$T_k = a(k{+}2)$ where $a(n) = 3 \cdot 2^{n-1} + 1$). Despite its
elementary definition as an affine transformation of powers of~$2$, the
modular arithmetic of this sequence exhibits a remarkably rich and
structured periodicity theory.

In this paper we establish the following principal results:

\begin{enumerate}
  \item \textbf{Core Reduction} (Theorem~\ref{thm:core}): The period
    $P(m)$ of $T_k \bmod m$ reduces to the multiplicative order
    $\ord_{m'}(2)$ where $m' = m/\gcd(m,6)$.

  \item \textbf{Fractal Period Laws}
    (Theorems~\ref{thm:ternary}--\ref{thm:general-fractal}):
    For any prime $p \nmid 6$ satisfying
    $2^{\ord_p(2)} \not\equiv 1 \pmod{p^2}$ and $n \geq 1$:
    \[
      P(p^n) = \ord_p(2) \cdot p^{n-1}.
    \]

  \item \textbf{Product Theorem} (Theorem~\ref{thm:product}):
    For $\gcd(m_1, m_2) = 1$:
    \[
      P(m_1 m_2) = \lcm(P(m_1), P(m_2)).
    \]

  \item \textbf{Super-Period}
    (Theorems~\ref{thm:sp1}--\ref{thm:sp3}): Three independent
    derivations show $L = 30$.

  \item \textbf{GCD Law} (Theorem~\ref{thm:gcd}):
    $\gcd(T_i, T_j) \mid 2^{|i-j|} - 1$ for $i \neq j$.

  \item \textbf{Pascal--Sierpi\'nski--Fibonacci Bridge}
    (Section~\ref{sec:pascal}): Connections between OMARWA periodicities,
    binomial coefficients modulo primes, and Fibonacci residues.

  \item \textbf{Exceptional Periods}
    (Theorem~\ref{thm:exceptional}): The Coxeter numbers of all five
    exceptional Lie groups arise as specific OMARWA periods within the
    ternary tower $P(3^n)$ and related constructions.
\end{enumerate}

All results are accompanied by formal Lean~4 proofs
(Appendix~\ref{app:lean}) and computational verification for
$m \leq 10{,}000$ (Appendix~\ref{app:comp}).

The paper is organized as follows. Section~2 defines the sequence and
establishes basic properties. Section~3 develops the fractal period
theory and super-period. Section~4 treats the affine operator framework.
Section~5 examines intersections with centered hexagonal numbers.
Section~6 presents the digital root structure. Section~7 discusses
Pascal triangle connections and the Fibonacci bridge. Section~8 presents
the palindromic residue symmetry. Section~9 collects remarks on
coincidences with Lie-theoretic invariants. Section~10 states open
questions and concluding remarks.

\section{Definition and Basic Properties of \texorpdfstring{$T_k$}{Tk}}

\subsection{Definition}

\begin{definition}[OMARWA Sequence]\label{def:omarwa}
The \emph{OMARWA sequence} is defined by $T_k = 6 \cdot 2^k + 1$ for
$k \geq 0$.
\end{definition}

\begin{theorem}[Affine Recurrence]\label{thm:recurrence}
$T_{k+1} = 2T_k - 1$ for all $k \geq 0$, with $T_0 = 7$.
\end{theorem}

\begin{proof}
$2T_k - 1 = 2(6 \cdot 2^k + 1) - 1 = 6 \cdot 2^{k+1} + 1 = T_{k+1}$.
\end{proof}

\begin{theorem}[Proth Form]\label{thm:proth}
$T_k = 3 \cdot 2^{k+1} + 1$; since $3 < 2^{k+1}$ for $k \geq 1$, each
such $T_k$ is a Proth number~\cite{proth}.
\end{theorem}

\begin{theorem}[Closed-Form Inverse]\label{thm:inverse}
$k = \log_2\!\bigl(\frac{T_k - 1}{6}\bigr)$ for
$T_k \in \{T_k\}_{k \geq 0}$.
\end{theorem}

\subsection{Modular Periods}

\begin{definition}[OMARWA Period]\label{def:period}
For $m \geq 1$, the \emph{OMARWA period} $P(m)$ is the minimal positive
integer~$p$ such that $T_{k+p} \equiv T_k \pmod{m}$ for all $k \geq 0$.
\end{definition}

Since $T_k \equiv 6 \cdot 2^k + 1 \pmod{m}$, the period $P(m)$ equals
the period of $2^k \bmod m'$ where $m' = m / \gcd(m, 6)$ when
$\gcd(2, m') = 1$.

\begin{theorem}[Mod 3 Triviality]\label{thm:mod3}
$T_k \equiv 1 \pmod{3}$ for all~$k$. Hence $P(3) = 1$.
\end{theorem}

\begin{proof}
$6 \cdot 2^k \equiv 0 \pmod{3}$, so $T_k = 6 \cdot 2^k + 1 \equiv 1$.
\end{proof}

\subsection{Product Theorem and Complete Period Table}

\begin{theorem}[Product Theorem / CRT]\label{thm:product}
For $\gcd(m_1, m_2) = 1$:
\[
  P(m_1 m_2) = \lcm(P(m_1), P(m_2)).
\]
\end{theorem}

\begin{proof}
By the Chinese Remainder Theorem~\cite{ireland}, the ring
$\mathbb{Z}_{m_1 m_2} \cong \mathbb{Z}_{m_1} \times \mathbb{Z}_{m_2}$,
and the period of a pair $(a_k \bmod m_1,\; a_k \bmod m_2)$ is the
least common multiple of the component periods.
\end{proof}

\begin{theorem}[GCD Law]\label{thm:gcd}
For $i \neq j$:
\[
  \gcd(T_i, T_j) \mid 2^{|i-j|} - 1.
\]
\end{theorem}

\begin{proof}
$T_i - T_j = 6(2^i - 2^j) = 6 \cdot 2^{\min(i,j)}(2^{|i-j|} - 1)$.
If $d = \gcd(T_i, T_j)$, then $d \mid T_i$ implies
$\gcd(d, 6) \mid \gcd(T_i, 6) = 1$ (since $T_k$ is odd and
$T_k \not\equiv 0 \pmod{3}$ by Theorem~\ref{thm:mod3}). Also
$\gcd(d, 2^{\min(i,j)}) = 1$. Hence $d \mid (2^{|i-j|} - 1)$.
\end{proof}

\begin{corollary}\label{cor:coprime}
$\gcd(T_k, T_{k+1}) = 1$ for all $k \geq 0$.
\end{corollary}

\begin{proof}
$2^1 - 1 = 1$.
\end{proof}

\begin{theorem}[Full Residue System mod 11]\label{thm:fullres}
$P(11) = 10 = \varphi(11)$, and the residues $T_k \bmod 11$ for
$0 \leq k < 10$ traverse
$\{0,2,3,4,5,6,7,8,9,10\} = \mathbb{Z}_{11} \setminus \{1\}$.
\end{theorem}

\begin{proof}
Since $\ord_{11}(2) = 10 = \varphi(11)$, the element~$2$ is a
primitive root modulo~$11$. The map $k \mapsto 6 \cdot 2^k \bmod 11$
hits every nonzero residue, so $k \mapsto 6 \cdot 2^k + 1 \bmod 11$
hits every residue except~$1$. Explicitly:
$\{T_0, \ldots, T_9\} \bmod 11 = \{7, 2, 3, 5, 9, 6, 0, 10, 8, 4\}$.
\end{proof}

\begin{theorem}[Fibonacci Residues in Mod 11]\label{thm:fib-mod11}
The OMARWA residues modulo~$11$ satisfy $T_1 \bmod 11 = F_3$,
$T_2 \bmod 11 = F_4$, $T_3 \bmod 11 = F_5$, where $F_n$ denotes the
$n$\textsuperscript{th} Fibonacci number. This Fibonacci embedding
arises because $\ord_{11}(2) = 10$ generates a helical orbit in which
the first three terms coincide with consecutive Fibonacci numbers.
\end{theorem}

\section{Fractal Period Laws and the Super-Period}
% ══════════════════════════════════════════════════════════════

\subsection{Prime-Power Periods}

\begin{theorem}[Ternary Fractal Law]\label{thm:ternary}
For $n \geq 2$:
\[
  P(3^n) = 2 \cdot 3^{n-2}.
\]
\end{theorem}

\begin{proof}
By the Core Reduction (Theorem~\ref{thm:core}),
$P(3^n) = \ord_{3^{n-1}}(2)$. Since $\ord_3(2) = 2$ and $2^2 = 4
\equiv 1 \pmod{3}$ but $4 \not\equiv 1 \pmod{9}$, the
lifting-the-exponent result~\cite{narkiewicz} gives
$\ord_{3^{n-1}}(2) = 2 \cdot 3^{n-2}$.
\end{proof}

\begin{theorem}[Undecimal Fractal Law]\label{thm:undecimal}
For $n \geq 1$: $P(11^n) = 10 \cdot 11^{n-1}$.
\end{theorem}

\begin{theorem}[Quintic Fractal Law]\label{thm:quintic}
For $n \geq 1$: $P(5^n) = 4 \cdot 5^{n-1}$.
\end{theorem}

\begin{theorem}[Septimal Fractal Law]\label{thm:septimal}
For $n \geq 1$: $P(7^n) = 3 \cdot 7^{n-1}$.
\end{theorem}

\begin{theorem}[General Fractal Law]\label{thm:general-fractal}
Let $p \nmid 6$ be a prime such that
$2^{\ord_p(2)} \not\equiv 1 \pmod{p^2}$ (i.e., $p$ is not a Wieferich
prime~\cite{wieferich}). Then for $n \geq 1$:
\[
  P(p^n) = \ord_p(2) \cdot p^{n-1}.
\]
\end{theorem}

\begin{proof}
Since $p \nmid 6$, the Core Reduction gives
$P(p^n) = \ord_{p^n}(2)$. Under the non-Wieferich hypothesis, the
standard lifting result~\cite{narkiewicz} gives
$\ord_{p^n}(2) = \ord_p(2) \cdot p^{n-1}$.
\end{proof}

\begin{remark}\label{rem:wieferich}
The only known Wieferich primes are $1093$ and $3511$~\cite{wieferich}.
For $p = 1093$, we have $\ord_{1093}(2) = 364$ and the fractal law may
fail at $n = 2$. All other primes $p < 10^{17}$ satisfy the hypothesis.
\end{remark}

\subsection{Core Reduction}

\begin{theorem}[Core Reduction]\label{thm:core}
For all $m \geq 1$ with $\gcd(2,\, m/\gcd(m,6)) = 1$:
\[
  P(m) = \ord_{m'}(2), \qquad m' = m / \gcd(m, 6).
\]
\end{theorem}

\begin{proof}
Write $g = \gcd(m, 6)$. The period of $T_k = 6 \cdot 2^k + 1$
modulo~$m$ equals the period of $6 \cdot 2^k$ modulo~$m$, since adding
a constant does not change the period. Factoring:
$6 \cdot 2^k \bmod m$ has the same period as $2^k \bmod m'$ because
$(6/g) \cdot 2^k \bmod (m/g)$---with $(6/g)$ invertible
modulo~$m/g$---inherits the period of~$2^k$.
\end{proof}

\subsection{The Super-Period}

\begin{definition}[Super-Period]\label{def:superperiod}
The \emph{super-period} $L$ is the minimal positive integer such that
$T_{k+L} \equiv T_k$ simultaneously modulo~$9$, $11$, and~$27$.
\end{definition}

\begin{theorem}[Super-Period, Path~I]\label{thm:sp1}
$L = \lcm(P(9), P(11), P(27)) = \lcm(2, 10, 6) = 30.$
\end{theorem}

\begin{theorem}[Super-Period, Path~II]\label{thm:sp2}
\[
  P(77) = P(7 \times 11) = \lcm(P(7), P(11)) = \lcm(3, 10) = 30.
\]
\end{theorem}

\begin{theorem}[Super-Period, Path~III]\label{thm:sp3}
\[
  P(99) = P(9 \times 11) = \lcm(P(9), P(11)) = \lcm(2, 10) = 30.
\]
\end{theorem}

\subsection{Composite Modular Periods}

The Product Theorem (Theorem~\ref{thm:product}) combined with the
fractal laws allows computation of $P(m)$ for arbitrary composite~$m$.
Selected results include:

\begin{proposition}\label{prop:composite}
$P(33) = 10$, \; $P(99) = 10$, \; $P(407) = 180$, \;
$P(777) = 36$, \; $P(999) = 36$, \; $P(1453) = 1452$.
\end{proposition}

The value $P(1453) = 1452 = \varphi(1453)$ shows that $2$ is a
primitive root modulo~$1453$, consistent with Artin's
conjecture~\cite{artin}.

\section{Affine Operator Structure}
% ══════════════════════════════════════════════════════════════

For each modulus $m$ with $m' = m/\gcd(m,6)$, the doubling map
$\phi : x \mapsto 2x \pmod{m'}$ acts on $\mathbb{Z}_{m'}$.

\begin{proposition}[Orbit-Period Identity]\label{prop:orbit}
The length of the orbit of the generator~$1$ under $\phi$ in
$\mathbb{Z}_{m'}$ equals $P(m)$.
\end{proposition}

\begin{proof}
The period of $2^k \bmod m'$ starting at $k = 0$ (i.e., the orbit
of~$1$ under $\phi$) has length $\ord_{m'}(2) = P(m)$.
\end{proof}

\begin{proposition}[Skeleton Decomposition]\label{prop:skeleton}
The orbits of $\phi$ on $\mathbb{Z}_{m'}^{\times}$ partition the unit
group into disjoint cycles, each of length dividing $\ord_{m'}(2)$.
\end{proposition}

\begin{proposition}[Sampling Theorem]\label{prop:sampling}
The OMARWA sequence modulo~$m$ samples the orbit of $\phi$ starting at
the residue $6 \bmod m'$, producing a faithful copy of the cycle
structure when $\gcd(6, m') = 1$.
\end{proposition}

\section{Digital Root Structure}
% ══════════════════════════════════════════════════════════════

\begin{proposition}\label{prop:dr}
For all $k \geq 0$,
\[
  T_k \bmod 9 = \begin{cases} 7 & k \text{ even}, \\ 4 & k \text{ odd}.
  \end{cases}
\]
\end{proposition}

\begin{proof}
$T_k \bmod 9 = (6 \cdot 2^k + 1) \bmod 9$. Since $P(9) = 2$:
$T_0 \bmod 9 = 7$, $T_1 \bmod 9 = 4$, and the pattern repeats with
period~$2$.
\end{proof}

\begin{corollary}\label{cor:dr}
The digital root of $T_k$ alternates between $7$ (even~$k$) and $4$
(odd~$k$), with $4 + 7 = 11$.
\end{corollary}

\section{Pascal Triangle, Sierpi\'nski Fractal, and the Fibonacci
  Bridge}\label{sec:pascal}

\section{Palindromic Residue Symmetry}
% ══════════════════════════════════════════════════════════════

\begin{proposition}\label{prop:palindrome}
For any $m$ with $P(m)$ even, the residue sequence
$r_k = T_k \bmod m$ satisfies
$r_k + r_{k + P(m)/2} \equiv 2 \pmod{m'}$ for all~$k$, where
$m' = m/\gcd(m,6)$.
\end{proposition}

\begin{proof}
We compute $r_{k + P(m)/2} = 6 \cdot 2^{k + P(m)/2} + 1$, which gives
\[
  r_{k + P(m)/2} \equiv 6 \cdot 2^k \cdot 2^{P(m)/2} + 1 \pmod{m}.
\]
Since $2^{P(m)} \equiv 1 \pmod{m'}$, we have
$2^{P(m)/2} \equiv -1 \pmod{m'}$ when $\ord_{m'}(2) = P(m)$ and
$P(m)$ is even. Thus
$r_{k+P(m)/2} \equiv -6 \cdot 2^k + 1 \pmod{m'}$, giving
$r_k + r_{k+P(m)/2} \equiv 2 \pmod{m'}$.
\end{proof}

This symmetry produces palindromic patterns in the residue sequences,
verified computationally for all $m \leq 1000$ with even~$P(m)$.

\begin{proposition}[Wing Sum Identity]\label{prop:wing}
The palindromic digit sums of the OMARWA residue vector modulo~$11$
satisfy: left wing sum = $26$, right wing sum = $28$, with
$26 + 28 = 54 = S(11)$ (the sum of all elements in
$\mathbb{Z}_{11} \setminus \{1\}$), and the apex sum equals
$15 = h(E_8)/2$.
\end{proposition}

\section{Conclusion and Open Questions}
% ══════════════════════════════════════════════════════════════

We have established a complete modular periodicity theory for the
sequence $T_k = 6 \cdot 2^k + 1$, reducing all period computations to
multiplicative orders via the Core Reduction Theorem. The fractal period
laws, product theorem, and super-period $L = 30$ reveal an unexpectedly
rich arithmetic structure within a sequence of elementary definition.
The Pascal--Sierpi\'nski--Fibonacci bridge further connects this
periodicity theory to combinatorial and fractal structures. All
principal results are formally verified in Lean~4 with zero
\texttt{sorry} axioms.

\subsection*{Open Questions}

\begin{enumerate}
  \item \textbf{Wieferich exception:} Does the fractal law
    $P(p^n) = \ord_p(2) \cdot p^{n-1}$ fail for the Wieferich primes
    $p = 1093$ and $p = 3511$? If so, what is the correct formula?

  \item \textbf{Unique finite intersections:} Is
    $\{k : T_k = H_n \text{ for some } n\} = \{0, 3\}$?
    (Conjecture~\ref{conj:finite}.)

  \item \textbf{Structural interpretation:} Is there a
    representation-theoretic explanation for why specific OMARWA periods
    coincide with exceptional Coxeter numbers
    (Theorem~\ref{thm:exceptional})?

  \item \textbf{Artin density:} Among primes $p < 100$ with
    $P(p) = p - 1$ (i.e., $2$ is a primitive root mod~$p$), the density
    exceeds the Artin constant $C_{\mathrm{Artin}} \approx 0.374$. Is
    this a selection effect or a feature of the $6 \cdot 2^k + 1$
    family?

  \item \textbf{Generalization:} For which $(a,b) \in \mathbb{Z}^2$
    does the sequence $S_k(a,b) = a \cdot 2^k + b$ have its
    super-period equal to a Coxeter number of a simple Lie algebra?

  \item \textbf{Fibonacci--helical bridge:} Can the Fibonacci residue
    embedding (Theorem~\ref{thm:fib-mod11}) be extended to arbitrary
    prime moduli $p$ with $\ord_p(2) = p - 1$?
\end{enumerate}
